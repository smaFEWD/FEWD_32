Find the Max Age

Exercise 4: Create a function that takes in ages from 4 users and computes the max age of all the users. Example output: Alert out "The oldest person in this group is: 45 years old." 

	function computeMaxAge() {
	  var firstAge = 32;
	  var secondAge = 44;
	  var thirdAge = 26;

	  var maxAge = -1;

	  if (firstAge > maxAge) {
	    maxAge = firstAge;
	  }

	  if (secondAge > maxAge) {
	    maxAge = secondAge;
	  }

	  if (thirdAge > maxAge) {
	    maxAge = thirdAge;
	  }
	  
	  console.log(maxAge);
	}
